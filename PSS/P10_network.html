df = raw[["userid", "user_mentions"]]
mentions = df.to_dict("records")

mentions[0].keys()

dict_keys(['userid', 'user_mentions'])

import re

edges = []
for relaiton in mentions:
    start, end = relaiton.values()
    end = re.sub("\[|\]", "", end)
    # end = end.replace("[|]", "")
    if end != "":
        if "," not in end:
            edges.append((start, end))
        if "," in end:
            for e in end.split(", "):
                edges.append((start, e))
